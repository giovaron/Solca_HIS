BEGIN
DECLARE
CURSOR PARAMETROS IS
SELECT * FROM PARAMETROS_PERFILES
WHERE PRFEMP_CODIGO = 'G' AND
      PRMROL_CODIGO IN (SELECT CODIGO
                        FROM PARAMETROS_ROLES@SOLCA) AND
      ESTADO_DE_DISPONIBILIDAD = 'D';
VVALOR NUMBER;
BEGIN
   FOR RPARAMETROS IN PARAMETROS LOOP
      SELECT VALOR INTO VVALOR
      FROM PARAMETROS_ROLES
      WHERE CODIGO = RPARAMETROS.PRMROL_CODIGO;
      INSERT INTO PARAMETROS_PERFILES@SOLCA(PRMROL_CODIGO,PRFEMP_CODIGO,PRMROL_EMP_CODIGO,
                                            PRFEMP_EMP_CODIGO,VALOR,ESTADO_DE_DISPONIBILIDAD)
      VALUES(RPARAMETROS.PRMROL_CODIGO,RPARAMETROS.PRFEMP_CODIGO,RPARAMETROS.PRMROL_EMP_CODIGO,
             RPARAMETROS.PRFEMP_EMP_CODIGO,VVALOR,RPARAMETROS.ESTADO_DE_DISPONIBILIDAD);
   END LOOP;
END;
END;
