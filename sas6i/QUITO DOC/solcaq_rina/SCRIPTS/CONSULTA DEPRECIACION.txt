SELECT ACF.ACTFSBGACTFJO_GRPACTFJO_CODIGO GRUPO,
       ACF.ACTFJOGNR_SBGACTFJO_CODIGO SUBGRUPO,
       ACF.CODIGO_PROPIO,TO_CHAR(ACF.FECHA_INGRESO,'DD/MM/RRRR') FECHA,
       SUBSTR(ACF.DESCRIPCION,1,100) NOMBRE,DPR.VALOR_INICIAL,
DPR.VALOR DEPRECIACION,DPR.VALOR_ACUMULADO,DPR.VALOR_INICIAL-(DPR.VALOR+DPR.VALOR_ACUMULADO) VALOR_RESIDUAL
FROM ACTIVOS_FIJOS_ESPECIFICOS ACF,DETALLES_DEPRECIACIONES DPR
WHERE ACF.SBGACTFJO_GRPACTFJO_EMP_CODIGO  = DPR.SBGACTFJO_GRPACTFJO_EMP_CODIGO AND
      ACF.ACTFSBGACTFJO_GRPACTFJO_CODIGO  = DPR.ACTFSBGACTFJO_GRPACTFJO_CODIGO AND
      ACF.ACTFJOGNR_SBGACTFJO_CODIGO      = DPR.ACTFACTFJOGNR_SBGACTFJO_CODIGO AND
      ACF.ACTFJOGNR_CODIGO                = DPR.ACTFJOESP_ACTFJOGNR_CODIGO AND
      ACF.CODIGO                          = DPR.ACTFJOESP_CODIGO AND
ORDER BY 1,2,3 
/
