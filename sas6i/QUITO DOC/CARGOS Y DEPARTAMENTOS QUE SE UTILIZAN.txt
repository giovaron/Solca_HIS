SELECT IC.DPR_ARA_CODIGO,IC.DPR_CODIGO,ARA.NOMBRE,DPR.NOMBRE,IC.TIPO,IC.CODIGO,DESCRIPCION
FROM CARGOS IC,DEPARTAMENTOS DPR,AREAS ARA
WHERE DPR.ARA_CODIGO=ARA.CODIGO AND
DPR.CODIGO=IC.DPR_CODIGO AND
ARA.CODIGO=IC.DPR_ARA_CODIGO AND
IC.ESTADO_DE_DISPONIBILIDAD='D' AND
(IC.TIPO,IC.CODIGO) IN
(SELECT CRG_TIPO,CRG_CODIGO
FROM DETALLES_FACTURA
WHERE FECHA>'01/01/2005')
ORDER BY 1,2,DESCRIPCION
