select CC.DESCRIPCION,er.apellidos||' '||er.nombres,
EMPCNTCST.pORCENTAJE
FROM EMPLEADOS_CENTROS_DE_COSTOS EMPCNTCST ,
empleados_roles er,
CENTROS_DE_COSTOS CC
WHERE 
(EMPCNTCST.CNTCST_EMP_CODIGO='CSI') 
and emprol_codigo=er.codigo
AND CC.TIPO=EMPCNTCST.CNTCST_TIPO
AND CC.AGRUPADOR=EMPCNTCST.CNTCST_AGRUPADOR
AND EMPCNTCST.ESTADO_DE_DISPONIBILIDAD='D'
/
