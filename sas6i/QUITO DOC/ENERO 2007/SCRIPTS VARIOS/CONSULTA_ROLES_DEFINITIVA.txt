SELECT
RTRIM(EMPROL.APELLIDOS)||' '||RTRIM(EMPROL.NOMBRES)
EMPLEADO,
CRGEMP.DESCRIPCION CARGO,
EMPROL.TERCERIZADORA BANCO,
EMPROL.CEDULA,
EMPROL.FECHA_DE_INGRESO FECHA_DE_INGRESO,
EMPROL.CODIGO EMPROL_CODIGO2,
DECODE(NVL(PRMROL.SECUENCIA_EN_ROL,0),0,'a ','z ')||PRMROL.NOMBRE ALIAS,
ROUND(MVMROL.VALOR,2) VALOR
FROM
CARGOS_EMPLEADOS CRGEMP,
EMPLEADOS_ROLES EMPROL,
MOVIMIENTOS_ROLES MVMROL,
PARAMETROS_ROLES PRMROL
WHERE
--( MVMROL.SECUENCIA_EN_ROL > 0 AND (MVMROL.VALOR > 0 OR MVMROL.IMPRIMIR_CERO = 'V'))  AND
'CSI' = MVMROL.ROLPGS_EMP_CODIGO AND
&NUMROL = MVMROL.ROLPGS_NUMERO AND
EMPROL.EMP_CODIGO = 'CSI'  AND
MVMROL.EMPROL_EMP_CODIGO = EMPROL.EMP_CODIGO AND
MVMROL.EMPROL_CODIGO = EMPROL.CODIGO AND
'CSI' = MVMROL.ROLPGS_EMP_CODIGO AND
&NUMROL = MVMROL.ROLPGS_NUMERO AND
EMPROL.CRGEMP_CODIGO = CRGEMP.CODIGO AND
EMPROL.CRGEMP_EMP_CODIGO = CRGEMP.EMP_CODIGO AND
'CSI'= MVMROL.ROLPGS_EMP_CODIGO AND
&NUMROL= MVMROL.ROLPGS_NUMERO AND
MVMROL.ESTADO!='A' AND
MVMROL.PRMROL_EMP_CODIGO=PRMROL.EMP_CODIGO AND
MVMROL.PRMROL_CODIGO=PRMROL.CODIGO
/
