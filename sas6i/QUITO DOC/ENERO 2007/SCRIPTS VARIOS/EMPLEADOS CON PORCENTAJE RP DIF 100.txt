SELECT ER.APELLIDOS||ER.NOMBRES,CC.DESCRIPCION,EC.PORCENTAJE
FROM EMPLEADOS_CENTROS_DE_COSTOS EC,EMPLEADOS_ROLES ER, CENTROS_DE_COSTOS CC
WHERE (ER.EMP_CODIGO,ER.CODIGO) IN (
SELECT  EMPROL_EMP_CODIGO,EMPROL_CODIGO
FROM EMPLEADOS_CENTROS_DE_COSTOS EMPCNTCST 
WHERE (EMPCNTCST.CNTCST_EMP_CODIGO='CSI')
GROUP BY EMPROL_EMP_CODIGO,EMPROL_CODIGO
HAVING SUM(PORCENTAJE)!=100)
AND EC.EMPROL_EMP_CODIGO=ER.EMP_CODIGO
AND EC.EMPROL_CODIGO=ER.CODIGO
AND CNTCST_EMP_CODIGO=CC.EMP_CODIGO
AND CNTCST_TIPO            =CC.TIPO
AND CNTCST_AGRUPADOR       =CC.AGRUPADOR
/
