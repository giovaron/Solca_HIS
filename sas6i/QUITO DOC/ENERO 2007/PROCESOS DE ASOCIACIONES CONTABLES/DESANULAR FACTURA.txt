
UPDATE CUENTAS_POR_COBRAR
SET ESTADO='CNC',TOTAL=4.00,SALDO=0,IVA=0,SALDO_IVA=0
WHERE NUMERO IN  (SELECT DISTINCT CTACBR_NUMERO
FROM FACTURAS WHERE CAJA='CJD' AND NUMERO IN (89297)) AND ESTADO='ANL';

SET ESTADO='CNC'
WHERE CTACBR_NUMERO= (SELECT DISTINCT CTACBR_NUMERO
FROM FACTURAS WHERE CAJA='CJD' AND NUMERO=89297) AND ESTADO='ANL';
