BEGIN
DECLARE CURSOR PESOS IS
SELECT EDAD,'M',CURVA,PC 
FROM CURVAS_DE_PC 
WHERE SEXO='F';

I NUMBER:=3;
J NUMBER:=2; -- 2 CUANDO TIENE UN SOLO HORARIO EL LUNES Y 3 CUANDO TIENE DOS HORARIOS EL LUNES
BEGIN
   FOR RPESOS IN PESOS LOOP
   BEGIN
      INSERT INTO CURVAS_DE_PC(EDAD,SEXO,CURVA,PC) 
                     VALUES(RPESOS.EDAD,'M',RPESOS.CURVA,RPESOS.PC);
   EXCEPTION
   WHEN OTHERS THEN
      DBMS_OUTPUT.PUT_LINE(SQLERRM);
   END;   
   END LOOP;
END;
COMMIT;
END;
/
