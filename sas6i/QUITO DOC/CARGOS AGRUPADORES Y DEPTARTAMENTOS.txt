SELECT AGRCNT_TIPO,AGRCNT_CODIGO,DESCRIPCION,DPR_ARA_CODIGO,DPR_CODIGO
FROM CARGOS
WHERE AGRCNT_CODIGO IS NOT NULL AND ESTADO_DE_DISPONIBILIDAD='D'
/

UPDATE CARGOS
SET AGRCNT_TIPO='5',AGRCNT_CODIGO='%'
WHERE AGRCNT_CODIGO IS NULL AND ESTADO_DE_DISPONIBILIDAD='D'
AND DPR_ARA_CODIGO='C' AND DPR_CODIGO='Y'
/
